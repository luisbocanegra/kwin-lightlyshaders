add_library(KWinQpaPlugin OBJECT)

target_sources(KWinQpaPlugin PRIVATE
    backingstore.cpp
    eglhelpers.cpp
    eglplatformcontext.cpp
    integration.cpp
    main.cpp
    offscreensurface.cpp
    platformcursor.cpp
    screen.cpp
    window.cpp
)

    IDENTIFIER KWIN_QPA

    DEFAULT_SEVERITY Critical
)

    Qt::CorePrivate
    Qt::GuiPrivate
    Freetype::Freetype # Must be after Qt6 platform support libs
